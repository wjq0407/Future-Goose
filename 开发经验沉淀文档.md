#### 1.2 流式响应处理
```typescript
export async function callAIAPIStream(
  messages: ChatMessage[],
  onChunk: (chunk: string) => void
): Promise<string> {
  const response = await fetch(API_CONFIG.baseURL + '/chat/completions', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      Authorization: `Bearer ${API_CONFIG.getApiKey()}`,
    },
    body: JSON.stringify({
      model: API_CONFIG.getModel(),
      messages,
      stream: true,
    }),
  });

  const reader = response.body.getReader();
  const decoder = new TextDecoder();
  let buffer = '';
  let fullContent = '';

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;

    buffer += decoder.decode(value, { stream: true });
    const lines = buffer.split('\n');
    buffer = lines.pop() || '';

    for (const line of lines) {
      const trimmed = line.trim();
      if (!trimmed || !trimmed.startsWith('data:')) continue;
      
      const dataStr = trimmed.slice(5).trim();
      if (dataStr === '[DONE]') continue;

      try {
        const parsed = JSON.parse(dataStr);
        const delta = parsed.choices?.[0]?.delta;
        if (delta?.content) {
          fullContent += delta.content;
          onChunk(delta.content);
        }
      } catch (e) {
        continue;
      }
    }
  }

  return fullContent;
}
```

## 3. 错误处理体系

#### 3.1 全局错误捕获
```typescript
export function initGlobalErrorHandling(): void {
  // 同步错误
  window.addEventListener('error', (event: ErrorEvent) => {
    event.preventDefault();
    const error = event.error || new Error(event.message);
    console.error('全局未捕获错误:', error);
    
    useToastStore.getState().error('发生错误', error.message);
  });

  // 异步错误
  window.addEventListener('unhandledrejection', (event: PromiseRejectionEvent) => {
    event.preventDefault();
    const reason = event.reason;
    console.error('未处理的 Promise 拒绝:', reason);
    
    // 特殊处理 API 错误
    if (reason?.errorType === 'API_KEY_NOT_SET') {
      useToastStore.getState().info('需要配置 API Key', '请点击设置按钮配置');
      return;
    }
    
    // 网络错误
    if (reason?.message?.includes('Failed to fetch')) {
      useToastStore.getState().error('网络连接失败', '请检查网络连接');
      return;
    }
    
    useToastStore.getState().error('操作失败', String(reason));
  });
}
```

#### 3.2 安全执行工具函数
```typescript
// 带降级值的异步执行
export function safeAsync<T>(
  promise: Promise<T>,
  fallbackValue?: T
): Promise<T | undefined> {
  return promise.catch((reason) => {
    console.error('安全执行错误:', reason);
    useToastStore.getState().error('操作失败', reason.message);
    return fallbackValue;
  });
}

// 带自定义错误处理的异步执行
export function handleAsyncError(
  promise: Promise<unknown>,
  errorHandler?: (error: Error) => void
): Promise<unknown> {
  return promise.catch((reason) => {
    const err = reason instanceof Error ? reason : new Error(String(reason));
    if (errorHandler) {
      errorHandler(err);
    } else {
      useToastStore.getState().error('操作失败', err.message);
    }
    throw err;
  });
}
```

#### 3.3 组件级错误边界
```typescript
class ErrorBoundary extends React.Component<Props, State> {
  state = { hasError: false, error: null };

  static getDerivedStateFromError(error: Error) {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error, info: React.ErrorInfo) {
    console.error('ErrorBoundary 捕获:', error, info);
  }

  render() {
    if (this.state.hasError) {
      return <ErrorFallback error={this.state.error} />;
    }
    return this.props.children;
  }
}
```

### 关键要点
- ✅ 全局错误捕获（window.onerror + unhandledrejection）
- ✅ 错误码 → 用户友好消息的映射机制
- ✅ Toast 通知系统统一展示
- ✅ ErrorBoundary 组件级错误隔离

export function initWebVitalsReport(callback: (metric: PerformanceMetric) => void): () => void {

  const a = document.createElement('a');
  a.href = url;
